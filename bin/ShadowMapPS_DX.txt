struct PS_INPUT
{
    float4 position     : SV_POSITION;
    float4 Dposition    : TEXCOORD0;
};

float4 ShadowPS(PS_INPUT input) : SV_Target
{
    //float depthVal = input.Dposition.z / input.Dposition.w;
    float depthVal = input.Dposition.z;
	depthVal /= 30000.0f;
    float4 colour = float4(depthVal, depthVal, depthVal, 1.0f);

    return colour;
}
